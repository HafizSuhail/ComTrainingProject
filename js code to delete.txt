function openModal(rowId) {
    // Set the row ID in a hidden field in the modal
    document.getElementById('rowIdInput').value = rowId;

    // Show the modal
    document.getElementById('deleteModal').style.display = 'block';
}

function closeModal() {
    // Close the modal
    document.getElementById('deleteModal').style.display = 'none';
}

function deleteRecord() {
    // Get the row ID from the hidden field
    var rowId = document.getElementById('rowIdInput').value;

    // Make an AJAX request to delete the record using the rowId
    // Replace 'yourDeleteEndpoint' with the actual endpoint to delete the record
    $.ajax({
        url: '/YourController/DeleteRecord/' + rowId,
        type: 'POST',
        success: function () {
            // Record deleted successfully
            // You can update the UI or perform any necessary actions here
            closeModal();
        },
        error: function () {
            // Handle the error, if any
            // You can display an error message or log the error
            closeModal();
        }
    });
}
